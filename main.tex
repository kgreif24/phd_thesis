% ============================================================
% UCI PhD Dissertation -- main.tex
% ============================================================
% \DocumentMetadata MUST come before \documentclass.
% It activates PDF/UA-1 conformance, sets the document language,
% and enables the LaTeX tagging infrastructure.
% Remove `uncompress` for final submission (produces a smaller PDF).
\DocumentMetadata{
  pdfversion = 1.7,
  pdfstandard = UA-1,
  lang        = en-US,
  uncompress,         % remove for final submission
}

\documentclass[12pt,fleqn]{ucithesis}

% ----------------------------------------------------------
% Math
% ----------------------------------------------------------
\usepackage{amsmath}
\usepackage{amsthm}
\usepackage{amssymb}

% ----------------------------------------------------------
% Graphics, tables, captions
% ----------------------------------------------------------
\usepackage[table]{xcolor}  % \definecolor, \rowcolor, \textcolor
\usepackage{array}
\usepackage{graphicx}
\usepackage{booktabs}      % professional-quality tables (toprule/midrule/bottomrule)
\usepackage{caption}
\usepackage{subcaption}    % subfigure environment
\usepackage{multirow}
\usepackage{tabularx}

% ----------------------------------------------------------
% Bibliography (biblatex + biber backend)
% ----------------------------------------------------------
\usepackage[backend=biber, style=numeric-comp, natbib=true, sorting=none]{biblatex}
\addbibresource{thesis.bib}

% ----------------------------------------------------------
% Feynman diagrams
% ----------------------------------------------------------
\usepackage{tikz}
\usepackage[compat=1.1.0]{tikz-feynman}

% ----------------------------------------------------------
% Miscellaneous layout
% ----------------------------------------------------------
\usepackage{relsize}
\usepackage[titletoc]{appendix}
\usepackage{parskip}

% ----------------------------------------------------------
% Hyperref (must be loaded late, before tagpdf and cleveref)
% plainpages=false       -- fixes duplicate page-counter warnings
% pdfborder={0 0 0}      -- no colored border around links
% unicode=true           -- required for PDF/UA
% pdfdisplaydoctitle     -- show title (not filename) in viewer titlebar
% ----------------------------------------------------------
\usepackage[
  plainpages         = false,
  pdfborder          = {0 0 0},
  unicode            = true,
  pdfdisplaydoctitle = true,
]{hyperref}

% ----------------------------------------------------------
% tagpdf -- PDF structure tagging for accessibility (PDF/UA-1)
% Must be loaded after hyperref.
%
% Troubleshooting:
%   - To temporarily disable:       \tagpdfsetup{activate-all=false}
%   - Around problematic content:   \tagpdfparaOff ... \tagpdfparaOn
%   - For final submission: remove `uncompress` (smaller file)
% ----------------------------------------------------------
\usepackage{tagpdf}
\tagpdfsetup{
  activate-all,
  uncompress,         % remove for final submission
}

% ----------------------------------------------------------
% Theorem environments
% ----------------------------------------------------------
\newtheorem{theorem}{\textsc{Theorem}}[chapter]
\newtheorem{definition}{\textsc{Definition}}[chapter]
\newtheorem{example}{\textsc{Example}}[chapter]

% ----------------------------------------------------------
% Thesis metadata (title, author, committee, abstract, etc.)
% ----------------------------------------------------------
\input{preliminaries}

% ----------------------------------------------------------
% Curriculum vitae (kept separate for easy maintenance)
% ----------------------------------------------------------
\input{vita}

% ----------------------------------------------------------
% PDF document metadata
% ----------------------------------------------------------
\hypersetup{
  pdftitle   = {\Thesistitle},
  pdfauthor  = {\Authorname},
  pdfsubject = {\Degreefield},
}

% ----------------------------------------------------------
% Custom style file (loads cleveref last -- cleveref MUST be
% the very last package loaded)
% ----------------------------------------------------------
\usepackage{thesis-style}

% Uncomment to build only specific chapters (preserves page numbers):
%\includeonly{chapters/chapter1}

% ============================================================
\begin{document}

% Preliminary pages (title, copyright, TOC, CV, abstract, etc.)
\preliminarypages

% Main chapters
\include{chapters/chapter1}
\include{chapters/chapter2}
\include{chapters/chapter3}
\include{chapters/chapter4}
\include{chapters/chapter5}
\include{chapters/chapter6}
\include{chapters/chapter7}

% Fix bibliography page number in TOC
\clearpage
\phantomsection

\printbibliography

% UCI manual: appendix figures/tables must NOT appear in LoF/LoT
\captionsetup[figure]{list=no}
\captionsetup[table]{list=no}

% Appendices (come after references per UCI manual §2.2.3)
\begin{appendices}
\include{appendices/appendix}
\end{appendices}

\end{document}
